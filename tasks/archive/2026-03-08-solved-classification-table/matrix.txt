   316	-- | Prune dead bind-parent entries from the canonical constraint.
   317	-- Used by Pipeline.hs.

   319	pruneBindParentsSolved (Solved eb@EquivBackend { ebCanonicalNodes = nodes, ebCanonicalGenNodes = genNodes0, ebCanonicalBindParents = bindParents0 }) =
   320	    let liveNodes = getNodeMap nodes
   321	        liveGens = getGenNodeMap genNodes0
   322	        liveRef ref =
   323	            case ref of
   324	                TypeRef nid -> IntMap.member (getNodeId nid) liveNodes
   325	                GenRef gid -> IntMap.member (genNodeKey gid) liveGens
   326	        liveChild childKey = liveRef (nodeRefFromKey childKey)

   328	            IntMap.filterWithKey
   329	                (\childKey (parentRef, _flag) ->

   332	                bindParents0
